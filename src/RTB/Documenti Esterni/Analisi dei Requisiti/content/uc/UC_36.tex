\subsubsection{UC\_36: Visualizza storico ordini completo}
\label{uc_36}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_36}
	\caption{UC\_36: Visualizza storico ordini completo}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin clicca il pulsante per visualizzare lo storico ordini
		\item L'Admin visualizza l'elenco di tutti gli ordini
		\item L'Admin visualizza il codice ordine $\rightarrow$ Vedi \hyperref[uc_36.1]{UC\_36.1}
		\item L'Admin visualizza la data dell'ordine $\rightarrow$ Vedi \hyperref[uc_36.2]{UC\_36.2}
		\item L'Admin visualizza lo username del Cliente $\rightarrow$ Vedi \hyperref[uc_36.3]{UC\_36.3}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item L'Admin è autenticato ed autorizzato
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item L'Admin si trova nella pagina web dedicata alla visualizzazione dello storico ordini
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_36.1]{UC\_36.1}
		\item \hyperref[uc_36.2]{UC\_36.2}
	\end{itemize}
	\item \textbf{Trigger}: L'Admin vuole visualizzare tutto lo storico degli ordini effettuati dai Clienti fino a quel momento
\end{itemize}
