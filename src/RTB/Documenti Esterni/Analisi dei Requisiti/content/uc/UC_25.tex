\subsubsection{UC\_25: Visualizza pagina performance}
\label{uc_25}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_25}
	\caption{UC\_25: Visualizza pagina performance}
\end{figure}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin accede alla pagina web di monitoraggio delle performance.
		\item L'Admin visualizza la durata media di risposta del chatbot $\rightarrow$ Vedi \hyperref[uc_25.1]{UC\_25.1}
		\item L'Admin visualizza la durata media di permanenza nella web-app $\rightarrow$ Vedi \hyperref[uc_25.2]{UC\_25.2}
		\item L'Admin visualizza le statistiche $\rightarrow$ Vedi \hyperref[uc_40]{UC\_40}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item L'Admin è autenticato ed autorizzato
		\item Il sistema è online
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Le metriche di performance sono state visualizzate
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_25.1]{UC\_25.1}
		\item \hyperref[uc_25.2]{UC\_25.2}
		\item \hyperref[uc_40]{UC\_40}
	\end{itemize}
	\item \textbf{Trigger}: L'Admin deve poter visualizzare le metriche di utilizzo del sistema, per valutarne le prestazioni e l'engagement degli utenti
\end{itemize}

\subsubsubsection{UC\_25.1: Visualizza durata media di risposta del chatbot}
\label{uc_25.1}
\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il sistema presenta la metrica relativa alla durata media di risposta del chatbot
		\item L'Admin visualizza la metrica
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item L'Admin si trova nella pagina delle performance $\rightarrow$ Vedi \hyperref[uc_25]{UC\_25}
		\item Il sistema ha dati sufficienti per calcolare la metrica
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item La metrica della durata media di risposta è stata correttamente visualizzata
	\end{itemize}
\end{itemize}
