\subsubsection{UC\_03: Autenticazione non riuscita}
\label{uc_03}

\begin{itemize}
    \item \textbf{Attore Principale}: Utente
    \item \textbf{Scenario Principale}:
    \begin{enumerate}
        \item L'Utente visualizza un messaggio di errore per username inesistente o pasword errata 
    \end{enumerate}
    \item \textbf{Precondizioni}:
    \begin{itemize}
        \item Il sistema è online
        \item L'Utente non è registrato nel sistema
        \item L'Utente ha inserito un username non valido oppure ha inserito una password non corretta
    \end{itemize}
    \item \textbf{Postcondizioni}:
    \begin{itemize}
        \item L'Utente visualizza il messaggio di errore \vr{Username o Password errata}
    \end{itemize}
\end{itemize}
