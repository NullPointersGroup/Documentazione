\subsubsection{UC\_03: Autenticazione non riuscita}
\label{uc_03}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Utente
    \item \textbf{Scenario Principale}:
    \begin{enumerate}
        \item L'Utente ha inserito uno username non valido oppure ha inserito una password non corretta
    \end{enumerate}
    \item \textbf{Precondizioni}:
    \begin{itemize}
        \item Il sistema è online
        \item L'Utente non è registrato nel sistema
    \end{itemize}
    \item \textbf{Postcondizioni}:
    \begin{itemize}
        \item L'Utente visualizza il messaggio di errore \vr{username o password errata}
    \end{itemize}
\end{itemize}
