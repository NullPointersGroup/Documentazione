\subsubsection{UC\_08: Reimposta password}
\label{uc_08}

\begin{figure}[H]
        \centering
        \includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_08}
        \caption{Estensioni e Inclusioni UC\_08: UC\_08.1, UC\_08.1.1, UC\_08.2, UC\_08.2.1, UC\_08.3, UC\_08.3.1}
\end{figure}

\begin{itemize}
    \item \textbf{Attore principale}: Cliente
    \item \textbf{Scenario principale}:
        \begin{enumerate}
            \item L'Utente entra nella sezione del profilo
            \item L'Utente clicca il tasto per reimpostare la password
            \item L'Utente inserisce la vecchia password $\rightarrow$ Vedi \hyperref[uc_08.1]{UC\_08.1}
            \item L'Utente inserisce la nuova password $\rightarrow$ Vedi \hyperref[uc_08.2]{UC\_08.2}
            \item L'Utente inserisce la conferma della nuova password $\rightarrow$ Vedi \hyperref[uc_08.3]{UC\_08.3}
        \end{enumerate}
    \item \textbf{Precondizioni}:
        \begin{itemize}
            \item Il sistema è online
            \item L'Utente è autenticato
            \item L'Utente conosce la precedente password
        \end{itemize}
    \item \textbf{Postcondizioni}:
        \begin{itemize}
            \item La password viene aggiornata nel sistema
        \end{itemize}
    \item \textbf{Inclusioni}: 
        \begin{itemize}
            \item \hyperref[uc_08.1]{UC\_08.1}
            \item \hyperref[uc_08.2]{UC\_08.2}
            \item \hyperref[uc_08.3]{UC\_08.3}
        \end{itemize}
    \item \textbf{User Story}: L'Utente vuole aggiornare le proprie credenziali di accesso
\end{itemize}
