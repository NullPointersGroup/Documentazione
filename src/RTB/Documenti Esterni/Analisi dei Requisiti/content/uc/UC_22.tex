\subsubsection{UC\_22: Annullamento disambiguazione}
\label{uc_22}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_22}
	\caption{UC\_22: Annullamento disambiguazione}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente decide di annullare il processo di disambiguazione
		\item Il Cliente visiona un messaggio di conferma per l'interruzione del processo di disambiguazione
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il processo di gestione ambiguità è attivo $\rightarrow$ Vedi \hyperref[uc_21]{UC\_21}
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il processo di disambiguazione viene annullato
		\item L'ordine rimane nello stato precedente all'ambiguità
	\end{itemize}
	\item \textbf{Trigger}: Il Cliente durante la chiarificazione di un prodotto ambiguo, vuole annullare l'operazione di disambiguazione
\end{itemize}

\begin{itemize}

	\item \textbf{Scenario principale}:

		\item Il Cliente invia il comando \verb|/annulla|
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item È attivo un processo di disambiguazione $\rightarrow$ Vedi \hyperref[uc_21]{UC\_21}
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il comando \verb|/annulla| è stato inviato
	\end{itemize}
\end{itemize}
