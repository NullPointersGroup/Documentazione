\subsubsection{UC\_37: Visualizza dettaglio ordine Admin}
\label{uc_37}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_37}
	\caption{UC\_37: Visualizza dettaglio ordine}
\end{figure}

\begin{itemize}
    \item \textbf{Attore Principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin seleziona un ordine specifico dalla lista dello storico
		\item Il sistema carica e visualizza la pagina di dettaglio dell'ordine
		\item L'Admin visualizza i dettagli dell'ordine, tra cui:
		\begin{itemize}
		    \item Lo username del Cliente $\rightarrow$ Vedi \hyperref[uc_37.1]{UC\_37.1}
		    \item I prodotti ordinati $\rightarrow$ Vedi \hyperref[uc_37.2]{UC\_37.2}
		    \item La data dell'ordine $\rightarrow$ Vedi \hyperref[uc_37.3]{UC\_37.3}
		    \item Il codice ordine $\rightarrow$ Vedi \hyperref[uc_37.4]{UC\_37.4}
		\end{itemize}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item  ed autorizzato
		\item Esiste almeno un ordine nello storico
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item I dettagli completi dell'ordine selezionato sono stati visualizzati
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_37.1]{UC\_37.1}
		\item \hyperref[uc_37.2]{UC\_37.2}
		\item \hyperref[uc_37.3]{UC\_37.3}
		\item \hyperref[uc_37.4]{UC\_37.4}
	\end{itemize}
	\item \textbf{Trigger}: L'Admin vuole ispezionare i dettagli completi di un ordine specifico
\end{itemize}
