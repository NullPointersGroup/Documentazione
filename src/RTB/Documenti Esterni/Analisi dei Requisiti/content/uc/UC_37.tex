\subsubsection{UC\_37: Visualizza dettaglio ordine Admin}
\label{uc_37}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_37}
	\caption{UC\_37: Visualizza dettaglio ordine}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Utente clicca su un ordine specifico
		\item L'Utente visualizza i dettagli di un ordine specifico
		\item L'Utente visualizza lo username del Cliente dell'ordine specifico $\rightarrow$ Vedi \hyperref[uc_37.1]{UC\_37.1}
		\item L'Utente visualizza i prodotti dell'ordine specifico $\rightarrow$ Vedi \hyperref[uc_37.2]{UC\_37.2}
		\item L'Utente visualizza la data dell'ordine specifico $\rightarrow$ Vedi \hyperref[uc_37.3]{UC\_37.3}
		\item L'Utente visualizza il numero ordine dell'ordine specifico $\rightarrow$ Vedi \hyperref[uc_37.4]{UC\_37.4}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item L'Admin è autenticato ed autorizzato
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il sistema mostra la pagina dedicata alla visualizzazione di un ordine
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_37.1]{UC\_37.1}
		\item \hyperref[uc_37.2]{UC\_37.2}
		\item \hyperref[uc_37.3]{UC\_37.3}
		\item \hyperref[uc_37.4]{UC\_37.4}
	\end{itemize}
	\item \textbf{User Story}: L'Admin vuole visualizzare i dettagli di un ordine specifico
\end{itemize}
