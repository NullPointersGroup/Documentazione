\subsubsection{UC\_45: Visualizza dettaglio ordine Cliente}
\label{uc_45}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_45}
	\caption{UC\_45: Visualizza dettaglio ordine Cliente}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente clicca su un proprio ordine specifico
		\item Il Cliente visualizza i dettagli di un proprio ordine specifico
		\item Il Cliente visualizza i prodotti del proprio ordine specifico $\rightarrow$ Vedi \hyperref[uc_45.1]{UC\_45.1}
		\item Il Cliente visualizza la data del proprio ordine specifico $\rightarrow$ Vedi \hyperref[uc_45.2]{UC\_45.2}
		\item Il Cliente visualizza il numero del proprio ordine specifico $\rightarrow$ Vedi \hyperref[uc_45.3]{UC\_45.3}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item Il Cliente è autenticato
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il Cliente visualizza la pagina dedicata alla visualizzazione di un ordine personale
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_45.1]{UC\_45.1}
		\item \hyperref[uc_45.2]{UC\_45.2}
		\item \hyperref[uc_45.3]{UC\_45.3}
	\end{itemize}
	\item \textbf{Trigger}: Il Cliente vuole visualizzare i dettagli di un proprio ordine specifico
\end{itemize}
