\subsubsection{UC\_45 Consultare i comandi disponibili}
\label{uc_45}
\begin{figure}[H]
    \centering
    \includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_45.png}
    \caption{UC\_45: Consultare i comandi disponibili}
\end{figure}
\begin{itemize}
    \item \textbf{Attore principale}: Utente
    \item \textbf{Scenario principale}: 
        \begin{enumerate}
            \item L'Utente apre il menù a scomparsa
            \item L'Utente visualizza l'elenco di comandi supportati
        \end{enumerate}
    \item \textbf{Precondizioni}: 
        \begin{itemize}
            \item Utente ha accesso alla chat
            \item Il sistema è attivo
        \end{itemize}
    \item \textbf{Postcondizioni}: 
        \begin{itemize}
            \item Menù dei comandi a scomparsa visualizzato nella parte superiore della chat
        \end{itemize}
    \item \textbf{User Story}: L'Utente vuole visualizzare l'elenco dei comandi che può eseguire
\end{itemize}
