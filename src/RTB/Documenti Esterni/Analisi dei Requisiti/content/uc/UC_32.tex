\subsubsection{UC\_32: Creazione nuovi utenti}
\label{uc_32}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_32&UC_33}
	\caption{UC\_32: Creazione nuovi utenti}
\end{figure}


\begin{itemize}
	\item \textbf{Attore principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin visualizza la pagina di gestione utenti $\rightarrow$ Vedi \hyperref[uc_32_1]{UC\_32.1}
		\item L'Admin inserisce il ruolo $\rightarrow$ Vedi \hyperref[uc_32_2]{UC\_32.2}
		\item L'Admin inserisce lo username $\rightarrow$ Vedi \hyperref[uc_32_3]{UC\_32.3}
		\item L'Admin inserisce l'email $\rightarrow$ Vedi \hyperref[uc_32_4]{UC\_32.4}
		\item Il Sistema mostra una notifica di creazione riuscita $\rightarrow$ Vedi \hyperref[uc_32_5]{UC\_32.5}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il Sistema è online
		\item L'Admin è autenticato
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il nuovo utente è stato creato nel Sistema
	\end{itemize}
	\item \textbf{Scenari alternativi}:
	\begin{itemize}
		\item La creazione utente fallisce e il Sistema mostra una notifica con spiegazione dell'errore $\rightarrow$ Vedi \hyperref[uc_33]{UC\_33}
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_32_1]{UC\_32.1}
		\item \hyperref[uc_32_2]{UC\_32.2}
		\item \hyperref[uc_32_3]{UC\_32.3}
		\item \hyperref[uc_32_4]{UC\_32.4}
		\item \hyperref[uc_32_5]{UC\_32.5}
	\end{itemize}
	\item \textbf{Estensioni}:
	\begin{itemize}
		\item \hyperref[uc_33]{UC\_33}
	\end{itemize}
	\item \textbf{User Story}: L'Admin vuole creare un nuovo utente inserendo i dati richiesti.
\end{itemize}
