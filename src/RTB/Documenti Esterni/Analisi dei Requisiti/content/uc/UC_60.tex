\subsubsection{UC\_60: Notifica conferma duplicazione ordine tramite chatbot}
\label{uc_60}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_60}
	\caption{UC\_60: Notifica conferma duplicazione ordine tramite chatbot}
\end{figure}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente visualizza una notifica di conferma della duplicazione dell'ordine
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item Il Cliente è autenticato
		\item La sessione chat del Cliente è attiva
		\item Il Cliente ha confermato la duplicazione dell'ordine tramite chatbot $\rightarrow$ Vedi \hyperref[uc_34]{UC\_34}
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il Cliente ha visualizzato la notifica di conferma di duplicazione dell'ordine
	\end{itemize}
	\item \textbf{Trigger}: Il Cliente vuole visualizzare una notifica di conferma di duplicazione dell'ordine se duplica un ordine tremite chatbot
\end{itemize}
