\subsubsection{UC\_60: Notifica conferma duplicazione ordine tramite chatbot}
\label{uc_60}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_60}
	\caption{UC\_60: Notifica conferma duplicazione ordine tramite chatbot}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il sistema mostra al Cliente una notifica di conferma della duplicazione
		\item Il Cliente visualizza la notifica
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il Cliente ha confermato la duplicazione dell'ordine tramite chatbot $\rightarrow$ Vedi \hyperref[uc_34]{UC\_34}
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il Cliente ha visualizzato la notifica di conferma
	\end{itemize}
	\item \textbf{Trigger}: Il Cliente vuole visualizzare una notifica di conferma in caso di duplicazione dell'ordine 
\end{itemize}
