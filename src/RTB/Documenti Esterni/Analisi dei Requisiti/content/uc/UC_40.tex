\subsubsection{UC\_40: Visualizza statistiche}
\label{uc_40}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_40}
	\caption{UC\_40: Visualizza pagina statistiche}
\end{figure}

\begin{itemize}
    \item \textbf{Attore Principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin visualizza la sezione statistiche dedicata compresa nella pagina delle performance
		\item L'Admin visualizza il numero di Utenti $\rightarrow$ Vedi \hyperref[uc_40.1]{UC\_40.1}
		\item L'Admin visualizza il numero di acquisti completati $\rightarrow$ Vedi \hyperref[uc_40.2]{UC\_40.2}
		\item L'Admin visualizza il modello utilizzato $\rightarrow$ Vedi \hyperref[uc_40.3]{UC\_40.3}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item L'Admin sta visualizzando la pagina delle performance $\rightarrow$ Vedi \hyperref[uc_25]{UC\_25}
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item L'Admin sta visualizzando la parte della pagina delle performance dedicata alle statistiche
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_40.1]{UC\_40.1}
		\item \hyperref[uc_40.2]{UC\_40.2}
		\item \hyperref[uc_40.3]{UC\_40.3}
	\end{itemize}
	\item \textbf{Trigger}: L'Admin vuole visualizzare le statistiche della piattaforma
\end{itemize}
