\subsubsection{UC\_01: Registrazione}
\label{uc_01}
\begin{figure}[H]
    \centering 
    \includegraphics[width=1\textwidth]{AnalisiRequisiti/UC_01.png}
\end{figure}
\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
    \item \textbf{Scenario Principale}: 
    \begin{itemize}
        \item Inserimento dello Username
        \item Inserimento della Password
        \item Inserimento della Password nel campo di "conferma password"
    \end{itemize}
    \item \textbf{Pre-Condizioni}:
    \begin{itemize}
        \item Il sistema è attivo
        \item Il cliente non è registrato nel Sistema
    \end{itemize}
    \item \textbf{Post-Condizioni}:
    \begin{itemize}
        \item L’utente è ora registrato ed è riconosciuto come Cliente
    \end{itemize}
    \item \textbf{Scenari Alternativi}:
    \begin{itemize}
        \item L'utente ha inserito uno username esistente
        \item L'utente ha inserito una password che non conforme
        \item L'utente ha inserito nel campo "conferma password" una password che non coincide con quella inserita precedentemente
    \end{itemize}
    \item \textbf{Inclusioni}:
    \begin{itemize}
        \item UC\_01.1(\ref{uc_01.1})
        \item UC\_01.1(\ref{uc_01.2})
        \item UC\_01.1(\ref{uc_01.3})
    \end{itemize}
    \item \textbf{Trigger}: Il cliente vuole registrarsi nel Sistema
\end{itemize}

\subsubsubsection{UC\_01.1: Inserimento Username}
\label{uc_01.1}
\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
    \item \textbf{Scenario Principale}:
    \begin{itemize}
        \item L'utente inserisce uno username con il quale vuole registrarsi
    \end{itemize}
    \item \textbf{Pre-Condizioni}: 
    \begin{itemize}
        \item Il sistema è attivo
        \item Il cliente non è registrato nel Sistema
        \item Lo username che l'utente inserisce non deve essere già presente nel sistema
    \end{itemize}
    \item \textbf{Post-Condizioni}:
    \begin{itemize}
        \item Username inserito, pronto per essere registrato
    \end{itemize}
    \item \textbf{Trigger}: Il cliente vuole registrarsi nel Sistema
\end{itemize}

\subsubsubsection{UC\_01.2: Inserimento Password}
\label{uc_01.2}
\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
    \item \textbf{Scenario Principale}:
    \begin{itemize}
        \item L'utente inserisce la password, che insieme allo username userà per autenticarsi
    \end{itemize}
    \item \textbf{Pre-Condizioni}: 
    \begin{itemize}
        \item Il sistema è attivo
        \item Il cliente non è registrato nel Sistema
        \item Il sistema non conosce la password
        \item 
    \end{itemize}
    \item \textbf{Post-Condizioni}:
    \begin{itemize}
        \item Il sistema riceve la password e ne conserva l'hash
    \end{itemize}
    \item \textbf{Trigger}: Il cliente vuole registrarsi nel Sistema
\end{itemize}

\subsubsubsection{UC\_01.3: Conferma Password}
\label{uc_01.3}
\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
    \item \textbf{Scenario Principale}:
    \begin{itemize}
        \item L'utente inserisce nuovamente la password nel campo "conferma password"
    \end{itemize}
    \item \textbf{Pre-Condizioni}:
    \begin{itemize}
        \item Il sistema è attivo
        \item Il cliente non è registrato nel sistema
        \item Il sistema conosce la password inserita in precedenza
    \end{itemize}
    \item \textbf{Post-Condizioni}:
    \begin{itemize}
        \item Password confermata, la registrazione può essere conclusa
    \end{itemize}
    \item \textbf{Trigger}: Il cliente vuole registrarsi nel Sistema
\end{itemize}
