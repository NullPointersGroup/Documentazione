\subsubsection{UC\_30: Esportazione storico ordini dei Clienti}
\label{uc_30}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_30}
	\caption{UC\_30: Esportazione storico ordini}
\end{figure}

\begin{itemize}
	\item \textbf{Attore$^G$ principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin richiede l'esportazione dello storico ordini in formato JSON$^G$
		\item Il sistema genera e rende disponibile il file JSON$^G$
		\item L'Admin scarica lo storico ordini in formato JSON$^G$
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item L'Admin è autenticato ed autorizzato
		\item Sono presenti ordini esportabili
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il file JSON$^G$ con lo storico ordini è stato generato e reso disponibile per il download
		\item L'Admin ha scaricato il file
	\end{itemize}
	\item \textbf{Scenari alternativi}:
	\begin{itemize}
		\item Si è verificato un errore nel download dello storico ordini $\rightarrow$ Vedi \hyperref[uc_31]{UC\_31}
	\end{itemize}
	\item \textbf{Estensioni}:
	\begin{itemize}
		\item \hyperref[uc_31]{UC\_31}
	\end{itemize}
	\item \textbf{Trigger}: L'Admin vuole esportare lo storico ordini in formato JSON$^G$
\end{itemize}
