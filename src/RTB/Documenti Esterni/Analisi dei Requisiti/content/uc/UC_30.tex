\subsubsection{UC\_30: Esportare storico ordini}
\label{uc_30}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_30}
	\caption{UC\_30: Esportazione storico ordini (JSON)}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin visualizza il pulsante per l'esportazione dello storico ordini
		\item L'Admin scarica lo storico ordini in formato JSON
		\item Il Sistema mostra una notifica di download riuscito $\rightarrow$ Vedi \hyperref[uc_30_1]{UC\_30.1}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il Sistema è online
		\item L'Admin è autenticato
		\item Sono presenti ordini esportabili
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item L'Admin ha scaricato correttamente lo storico ordini in formato JSON
	\end{itemize}
	\item \textbf{Scenari alternativi}:
	\begin{itemize}
		\item L'Admin applica almeno un filtro prima dell'esportazione $\rightarrow$ Vedi \hyperref[uc_36.1]{UC\_36.1}
		\item Si verifica un errore nello scaricamento dello storico ordini (JSON) $\rightarrow$ Vedi \hyperref[uc_31]{UC\_31}
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_30_1]{UC\_30.1}
	\end{itemize}
	\item \textbf{Estensioni}:
	\begin{itemize}
		\item \hyperref[uc_36.1]{UC\_36.1}
		\item \hyperref[uc_31]{UC\_31}
	\end{itemize}
	\item \textbf{User Story}: L'Admin vuole esportare lo storico ordini in formato JSON.
\end{itemize}

\subsubsubsection{UC\_30.1: Visualizzazione notifica (download riuscito)}
\label{uc_30_1}

\begin{itemize}
	\item \textbf{Attore principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin visualizza una notifica di download riuscito
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item L'Admin ha scaricato lo storico ordini $\rightarrow$ Vedi \hyperref[uc_30]{UC\_30}
		\item Il file JSON è stato generato correttamente
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item La notifica di download riuscito è visualizzata
	\end{itemize}
	\item \textbf{User Story}: L'Admin vuole ricevere una conferma visiva del corretto completamento del download.
\end{itemize}
