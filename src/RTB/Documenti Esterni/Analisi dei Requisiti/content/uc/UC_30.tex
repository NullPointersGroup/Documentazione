\subsubsection{UC\_30: Esportare storico ordini}
\label{uc_30}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_30&UC_31}
	\caption{UC\_30: Esportazione storico ordini (JSON)}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin richiede l'esportazione dello storico ordini in formato JSON
		\item L'Admin scarica lo storico ordini in formato JSON
		\item Il Sistema mostra una notifica di download riuscito $\rightarrow$ Vedi \hyperref[uc_30_2]{UC\_30.2}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il Sistema è online
		\item L'Admin è autenticato
		\item Sono presenti ordini esportabili
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item L'Admin ha scaricato un file JSON contenente lo storico ordini
	\end{itemize}
	\item \textbf{Scenari alternativi}:
	\begin{itemize}
		\item L'Admin applica almeno un filtro prima dell'esportazione $\rightarrow$ Vedi \hyperref[uc_30_1]{UC\_30.1}
		\item Si verifica un errore nel caricamento dello storico ordini in formato JSON $\rightarrow$ Vedi \hyperref[uc_31]{UC\_31}
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_30_2]{UC\_30.2}
	\end{itemize}
	\item \textbf{Estensioni}:
	\begin{itemize}
		\item \hyperref[uc_30_1]{UC\_30.1}
		\item \hyperref[uc_31]{UC\_31}
	\end{itemize}
	\item \textbf{User Story}: L'Admin vuole esportare lo storico ordini in formato JSON.
\end{itemize}
