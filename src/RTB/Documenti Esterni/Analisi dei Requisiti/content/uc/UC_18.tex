\subsubsection{UC\_18: Errore rimozione articolo dal carrello}
\label{uc_18}

\begin{itemize}
    \item \textbf{Attore principale}: Utente
    \item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Utente riceve un messaggio di errore rimozione articolo dal carrello
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item L'Utente aveva richiesto la rimozione di articoli dal carrello
		\item Il carrello è vuoto
		\item Il sistema ha avuto un calo di rete
		\item Gli articoli di cui viene richiesta la rimozione non sono presenti nel carrello
    \end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item La rimozione degli articoli dal carrello fallisce
	\end{itemize}
	\item \textbf{User Story}: L'Utente voleva rimuovere degli articoli dal carrello, ma l'eliminazione non è andata a buon fine
\end{itemize}
