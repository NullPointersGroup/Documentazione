\subsubsection{UC\_47: Impostazione filtri Admin}
\label{uc_47}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_47}
	\caption{UC\_47: Impostazione filtri Admin}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Admin
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Admin accede alle impostazioni dei filtri in una pagina
		\item L'Admin imposta i filtri secondo le proprie preferenze
		\item L'Admin imposta il filtro Cliente $\rightarrow$ Vedi \hyperref[uc_47.1]{UC\_47.1}
		\item L'Admin imposta il filtro data $\rightarrow$ Vedi \hyperref[uc_47.2]{UC\_47.2}
		\item L'Admin imposta il filtro prodotti $\rightarrow$ Vedi \hyperref[uc_47.3]{UC\_47.3}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item L'Admin è autenticato ed autorizzato
		\item L'Admin si trova in una pagina che supporta i filtri
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item L'Admin ha configurato i filtri secondo le proprie preferenze
		\item L'Admin visualizza la pagina aggiornata con i filtri applicati
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_47.1]{UC\_47.1}
		\item \hyperref[uc_47.2]{UC\_47.2}
		\item \hyperref[uc_47.3]{UC\_47.3}
	\end{itemize}
	\item \textbf{Trigger}: L'Admin vuole personalizzare la visualizzazione dei dati applicando filtri specifici
\end{itemize}
