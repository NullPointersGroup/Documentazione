\subsubsection{UC\_20: Avviso carrello vuoto}
\label{uc_20}

\begin{itemize}
	\item \textbf{Attore principale}: Utente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Utente digita il comando \texttt{/carrello}
		\item L'Utente visualizza un avviso riguardante il carrello vuoto
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item L'Utente ha digitato il comando \texttt{/carrello}
		\item Il carrello è vuoto
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item L'Utente visualizza un avviso che comunica che il carrello è vuoto
	\end{itemize}
	\item \textbf{User Story}: L'Utente prova a visualizzare il riepilogo del carrello che però è vuoto, di conseguenza viene informato del fatto che non sia presente alcun articolo nel carrello
\end{itemize}
