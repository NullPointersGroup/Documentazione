\subsubsection{UC\_39: Errore duplicazione ordine}
\label{uc_39}

\begin{itemize}
	\item \textbf{Attore principale}: Utente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Utente riceve un messaggio di errore duplicazione ordine
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema ha avuto un calo di rete
		\item L'Utente aveva confermato la duplicazione dell'ordine
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item La duplicazione dell'ordine non è riuscita
	\end{itemize}
	\item \textbf{User Story}: L'Utente voleva duplicare un ordine, ma per un problema di rete o interno al sistema non è riuscito, di conseguenza riceve un messaggio di errore duplicazione ordine
\end{itemize}
