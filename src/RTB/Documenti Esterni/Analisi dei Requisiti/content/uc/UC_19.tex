\subsubsection{UC\_19: Visualizza anteprima ordine}
\label{uc_19}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_19}
	\caption{UC\_19: Visualizza anteprima ordine}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Utente
    \item \textbf{Scenario principale}:
	\begin{enumerate}
		\item L'Utente digita il comando testuale \verb|/carrello| nella chat
		\item L'Utente visualizza il riepilogo del carrello
		\item L'Utente visualizza il prezzo dei prodotti presenti nel carrello $\rightarrow$ Vedi \hyperref[uc_19.1]{UC\_19.1}
		\item L'Utente visualizza il nome dei prodotti presenti nel carrello $\rightarrow$ Vedi \hyperref[uc_19.2]{UC\_19.2}
		\item L'Utente visualizza la quantità dei prodotti presenti nel carrello $\rightarrow$ Vedi \hyperref[uc_19.3]{UC\_19.3}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item La sessione chat dell'Utente è attiva
		\item Sono presenti articoli nel carrello
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item L'Utente visualizza correttamente l'anteprima del proprio carrello
	\end{itemize}
	\item \textbf{Scenari alternativi}:
	\begin{itemize}
		\item Il carrello è vuoto, l'Utente visualizza un messaggio di avviso specifico. $\rightarrow$ Vedi \hyperref[uc_20]{UC\_20}
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_19.1]{UC\_19.1}
		\item \hyperref[uc_19.2]{UC\_19.2}
		\item \hyperref[uc_19.3]{UC\_19.3}
	\end{itemize}
	\item \textbf{Estensioni}:
	\begin{itemize}
		\item \hyperref[uc_20]{UC\_20}
	\end{itemize}
    \item \textbf{User Story}:
	L'Utente vuole visualizzare il riepilogo dell'ordine corrente, comprensivo di nomi, quantità e prezzi degli articoli, per verificarne il contenuto o effettuare eventuali modifiche.
\end{itemize}
