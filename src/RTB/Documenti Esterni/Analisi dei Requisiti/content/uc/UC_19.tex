\subsubsection{UC\_19: Visualizza anteprima ordine}
\label{uc_19}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_19}
	\caption{UC\_19: Visualizza anteprima ordine}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Cliente
    \item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente invia il comando \verb|/carrello| nella chat $\rightarrow$ Vedi \hyperref[uc_19.1]{UC\_19.1}
		\item Il Cliente visualizza il riepilogo del carrello $\rightarrow$ Vedi \hyperref[uc_19.2]{UC\_19.2}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item La sessione chat del Cliente è attiva
		\item Sono presenti articoli nel carrello
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il Cliente visualizza correttamente l'anteprima del proprio carrello
	\end{itemize}
	\item \textbf{Scenari alternativi}:
	\begin{itemize}
		\item Il carrello è vuoto, il Cliente visualizza un messaggio di avviso specifico. $\rightarrow$ Vedi \hyperref[uc_20]{UC\_20}
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_19.1]{UC\_19.1}
		\item \hyperref[uc_19.2]{UC\_19.2}
	\end{itemize}
	\item \textbf{Estensioni}:
	\begin{itemize}
		\item \hyperref[uc_20]{UC\_20}
	\end{itemize}
    \item \textbf{Trigger}:
	Il Cliente vuole visualizzare il riepilogo dell'ordine corrente, comprensivo di nomi, quantità e prezzi degli articoli, per verificarne il contenuto o effettuare eventuali modifiche.
\end{itemize}
