\subsubsection{UC\_55: Pulsante invio disabilitato}
\label{uc_55}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
    \item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente visualizza il pulsante per l'invio dell'ordine disabilitato
    \end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item Il Cliente è autenticato
		\item La sessione chat del Cliente è attiva
		\item Non sono presenti articoli nel carrello
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il Cliente visualizza il pulsante di invio dell'ordine disabilitato
	\end{itemize}
    \item \textbf{Trigger}: Il Cliente vuole inviare l'ordine ma non sono presenti articoli nel carrello
\end{itemize}
