\subsubsection{UC\_11: Inserimento ordine via immagine}
\label{uc_11}

\begin{figure}[H]
        \centering
        \includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_11}
        \caption{UC\_11: Inserimento ordine via immagine}
\end{figure}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
    \item \textbf{Scenario principale}:
        \begin{enumerate} 
            \item Il Cliente inserisce un ordine tramite immagine
            \item Il Cliente scatta una foto con la fotocamera $\rightarrow$ Vedi \hyperref[uc_11.1]{UC\_11.1}
            \item Il Cliente seleziona un'immagine dal dispositivo $\rightarrow$ Vedi \hyperref[uc_11.2]{UC\_11.2}
        \end{enumerate}
    \item \textbf{Precondizioni}:
        \begin{itemize}
            \item Il sistema è online
            \item Il Cliente è autenticato
            \item Il Cliente visualizza il pulsante per allegare le immagini
            \item Il Cliente visualizza il pulsante d'invio
        \end{itemize}
    \item \textbf{Postcondizioni}:
        \begin{itemize}
            \item L'immagine viene archiviata
            \item L'immagine viene inviata all'OCR$^G$
        \end{itemize}
    \item \textbf{Scenari alternativi}:
        \begin{itemize}
            \item Il Cliente riceve un messaggio di errore per immagine non conforme $\rightarrow$ Vedi \hyperref[uc_12]{UC\_12}
        \end{itemize}
    \item \textbf{Estensioni}:
    \begin{itemize}
            \item \hyperref[uc_12]{UC\_12}
        \end{itemize}
    \item \textbf{Generalizzazioni}:
        \begin{itemize}
            \item \hyperref[uc_11.1]{UC\_11.1}
            \item \hyperref[uc_11.2]{UC\_11.2}
        \end{itemize}
    \item \textbf{Trigger}: Il Cliente vuole inserire un ordine scattando una foto o selezionando un'immagine dal proprio dispositivo
\end{itemize}


\subsubsubsection{UC\_11.1: Scatto foto con fotocamera}
\label{uc_11.1}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
    \item \textbf{Scenario principale}:
        \begin{enumerate}
            \item Il Cliente ha premuto il pulsante per utilizzare la fotocamera del dispositivo
        \end{enumerate}
    \item \textbf{Precondizioni}:
        \begin{itemize}
            \item La fotocamera è disponibile e attivabile
        \end{itemize}
    \item \textbf{Postcondizioni}: 
        \begin{itemize}

            \item Il sistema invia l'immagine appena scattata all'OCR$^G$ 
        \end{itemize}
    \item \textbf{Trigger}: Il Cliente vuole inserire un ordine scattando una foto
\end{itemize}


\subsubsubsection{UC\_11.2: Selezionare immagine da dispositivo}
\label{uc_11.2}

L'UC\_11.2 ha 3 ulteriori generalizzazioni.
\begin{figure}[H]
        \centering
        \includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_11.2}
        \caption{Generalizzazioni UC\_11.2: UC\_11.2.1, UC\_11.2.2, UC\_11.2.3}
\end{figure}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
    \item \textbf{Scenario principale}:
        \begin{enumerate}
            \item Il Cliente ha premuto il pulsante per selezionare un file dal dispositivo
            \item Il Cliente seleziona l'immagine
        \end{enumerate}
    \item \textbf{Precondizioni}:
        \begin{itemize}
            \item Il Cliente visualizza il tasto per la selezione dei media da dispositivo
        \end{itemize}
    \item \textbf{Postcondizioni}: 
        \begin{itemize}
            \item L'immagine viene archiviata
            \item Il sistema invia l'immagine selezionata all'OCR$^G$ 
        \end{itemize}
    \item \textbf{Generalizzazioni}:
        \begin{itemize}
            \item \hyperref[uc_11.2.1]{UC\_11.2.1}
            \item \hyperref[uc_11.2.2]{UC\_11.2.2}
            \item \hyperref[uc_11.2.3]{UC\_11.2.3}
        \end{itemize}
    \item \textbf{Trigger}: Il Cliente vuole inserire un ordine mediante un'immagine presente nel dispositivo
\end{itemize}
