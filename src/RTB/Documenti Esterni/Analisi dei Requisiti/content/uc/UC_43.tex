	\item \textbf{Scenario principale}: 
	\begin{enumerate}
		\item Il Cliente scrive nella chat \vr{/} per visualizzare i comandi

	\item \textbf{Precondizioni}: 
	\begin{itemize}
		\item Il Cliente ha accesso alla chat
		\item Il sistema è attivo
	\end{itemize}
	\item \textbf{Postcondizioni}: 
	\begin{itemize}
		\item Il Cliente visualizza i comandi disponibili
	\end{itemize}
	\item \textbf{Inclusioni}:

	\end{itemize}

\end{itemize}
