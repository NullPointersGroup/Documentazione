\subsubsection{UC\_43 Visualizza comandi inline} 
\label{uc_43}
\begin{figure}[H]
    \centering
    \includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_43.png}
    \caption{UC\_43: Visualizzare i comandi inline}
\end{figure}
\begin{itemize}
    \item \textbf{Attore principale}: Utente
    \item \textbf{Scenario principale}: 
        \begin{enumerate}
            \item L'Utente visualizza un prompt inline relativo alla visualizzazione dei comandi nella chat.
        \end{enumerate}
    \item \textbf{Precondizioni}: 
        \begin{itemize}
            \item Utente ha accesso alla chat
            \item Il sistema è attivo
        \end{itemize}
    \item \textbf{Postcondizioni}: 
        \begin{itemize}
            \item Menù dei comandi a comparsa visualizzato nella parte superiore della chat
        \end{itemize}
    \item \textbf{Inclusioni}:
        \begin{itemize}
		    \item \hyperref[uc_43.1]{UC\_43.1}
        \end{itemize}
    \item \textbf{User Story}: L'Utente vuole visualizzare l'elenco dei comandi che può eseguire
\end{itemize}
