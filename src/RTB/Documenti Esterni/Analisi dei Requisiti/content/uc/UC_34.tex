\subsubsection{UC\_34: Duplicazione ordine tramite chatbot}
\label{uc_34}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_34}
	\caption{UC\_34: Duplicazione ordine tramite chatbot}
\end{figure}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente richiede la duplicazione di un ordine tramite chatbot
		\item Il sistema identifica e propone l'ordine da duplicare
		\item Il Cliente conferma la duplicazione $\rightarrow$ Vedi \hyperref[uc_34.2]{UC\_34.2}
		\item Il sistema duplica l'ordine
		\item Il sistema mostra una notifica di conferma $\rightarrow$ Vedi \hyperref[uc_34.3]{UC\_34.3}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item Il Cliente è autenticato
		\item La sessione chat del Cliente è attiva
		\item Esiste almeno un ordine precedente del Cliente
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Una copia dell'ordine selezionato è stata creata nel carrello
	\end{itemize}
	\item \textbf{Scenari alternativi}:
	\begin{itemize}
		\item L'ordine non esiste e il sistema mostra un errore di duplicazione $\rightarrow$ Vedi \hyperref[uc_35]{UC\_35}
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_34.1]{UC\_34.1}
		\item \hyperref[uc_34.2]{UC\_34.2}
		\item \hyperref[uc_34.3]{UC\_34.3}
	\end{itemize}
	\item \textbf{Estensioni}:
	\begin{itemize}
		\item \hyperref[uc_35]{UC\_35}
	\end{itemize}
	\item \textbf{Trigger}: Il Cliente vuole duplicare un ordine precedente tramite chatbot
\end{itemize}

\subsubsubsection{UC\_34.1: Visualizza riepilogo ordine}
\label{uc_34.1}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
    \item \textbf{Scenario principale}:
    \begin{enumerate}
        \item Il Cliente visualizza la lista di articoli presenti nell'ordine da duplicare $\rightarrow$ Vedi \hyperref[uc_34.1.1]{UC\_34.1.1}
    \end{enumerate}
    \item \textbf{Precondizioni}:
    \begin{itemize}
        \item L'ordine da duplicare esiste.
        \item Il Cliente ha richiesto la duplicazione di un ordine tramite chatbot e il sistema sta mostrando il riepilogo dell'ordine da duplicare
    \end{itemize}
    \item \textbf{Postcondizioni}:
    \begin{itemize}
        \item Il Cliente visualizza gli articoli presenti nell'ordine da duplicare tramite riepilogo
    \end{itemize}
    \item \textbf{Inclusioni}:
	\begin{itemize}
        \item \hyperref[uc_34.1.1]{UC\_34.1.1}
	\end{itemize}
    \item \textbf{Trigger}: Il Cliente vuole duplicare l'ordine ma prima è necessario che verifichi gli articoli dell'ordine da duplicare
\end{itemize}

\subsubsubsection{UC\_34.2: Conferma duplicazione}
\label{uc_34.2}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente conferma la duplicazione dell'ordine proposto
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il Cliente ha richiesto la duplicazione di un ordine e il sistema ha proposto un ordine valido $\rightarrow$ Vedi \hyperref[uc_34]{UC\_34}
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item La duplicazione dell'ordine è confermata dal Cliente
	\end{itemize}
	\item \textbf{Trigger}: Il sistema richiede conferma per procedere con la duplicazione
\end{itemize}

\subsubsubsection{UC\_34.3: Notifica conferma}
\label{uc_34.3}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il sistema mostra al Cliente una notifica di conferma della duplicazione
		\item Il Cliente visualizza la notifica
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il Cliente ha confermato la duplicazione $\rightarrow$ Vedi \hyperref[uc_34.2]{UC\_34.2}
		\item Il sistema ha completato la duplicazione dell'ordine
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item La notifica di conferma è stata visualizzata
	\end{itemize}
	\item \textbf{Trigger}: Il sistema ha completato con successo la duplicazione dell'ordine
\end{itemize}
