\begin{itemize}
    \item \textbf{Attore principale}: Utente
    \item \textbf{Scenario principale}:
        \begin{itemize}
            \item L'Utente inserisce un ordine tramite input testuale

        \begin{itemize}
            \item Gli articoli vengono aggiunti al carrello
        \end{itemize}
    \item \textbf{Scenari alternativi}:
        \begin{itemize}
            \item L'utente riceve un messaggio di errore di aggiunta articolo al carrello. $\rightarrow$ Vedi [UC\_16, §\ref{uc_16}]

    \begin{itemize}
            \item $[$UC\_16, §\ref{uc_16}$]$
        \end{itemize}
