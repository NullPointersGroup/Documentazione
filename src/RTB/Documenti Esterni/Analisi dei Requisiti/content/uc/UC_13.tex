\subsubsection{UC\_13: Inserimento ordine via audio}
\label{uc_13}
\begin{figure}[H]
        \centering
        \includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_13}
        \caption{UC\_13: Inserimento ordine via audio}
\end{figure}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
    \item \textbf{Scenario principale}:
        \begin{enumerate}
            \item Il Cliente inserisce un ordine tramite audio
        \end{enumerate}
    \item \textbf{Precondizioni}:
        \begin{itemize}
            \item Il sistema è online
            \item Il Cliente è autenticato
            \item Il Cliente visualizza il pulsante per allegare i file audio
            \item Il Cliente visualizza il pulsante per registrare un messaggio audio
            \item Il Cliente visualizza il pulsante d'invio
        \end{itemize}
    \item \textbf{Postcondizioni}:
        \begin{itemize}
            \item L'audio viene archiviato
            \item L'audio viene inviato allo Speech-to-Text
        \end{itemize}
    \item \textbf{Scenari alternativi}:
        \begin{itemize}
            \item Il Cliente riceve un messaggio di errore per il caricamento dell'audio $\rightarrow$ Vedi \hyperref[uc_14]{UC\_14}
        \end{itemize}
    \item \textbf{Estensioni}:
    \begin{itemize}
            \item \hyperref[uc_14]{UC\_14}
        \end{itemize}
    \item \textbf{Specializzazioni}:
        \begin{itemize}
            \item \hyperref[uc_13.1]{UC\_13.1}
            \item \hyperref[uc_13.2]{UC\_13.2}
        \end{itemize}
    \item \textbf{Trigger}: Il Cliente vuole inserire un ordine registrando un audio o selezionandone uno dal proprio dispositivo
\end{itemize}


\subsubsubsection{UC\_13.1: Registrare file audio}
\label{uc_13.1}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
    \item \textbf{Scenario principale}:
        \begin{enumerate}
            \item Il Cliente ha premuto il pulsante per registrare un audio
            \item Il Cliente registra l'audio
            \item Viene visualizzato il timer del tempo rimanente
            \item Viene visualizzato il pulsante per fermare la registrazione
        \end{enumerate}
    \item \textbf{Precondizioni}:
        \begin{itemize}
            \item Il Cliente visualizza il pulsante per la registrazione di un audio
        \end{itemize}
    \item \textbf{Postcondizioni}: 
        \begin{itemize}
            \item L'audio viene archiviato
            \item L'audio viene inviato allo Speech-to-Text
        \end{itemize}
    \item \textbf{Trigger}: Il Cliente vuole inserire un ordine registrando un messaggio audio
\end{itemize}

\end{itemize}
