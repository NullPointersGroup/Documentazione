\subsubsection{UC\_41 Visualizzare elenco dei comandi in un'altra pagina}
\label{uc_41}
\begin{figure}[H]
    \centering
    \includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_41.png}
    \caption{UC\_41: Visualizzare elenco dei comandi in un'altra pagina}
\end{figure}
\begin{itemize}
    \item \textbf{Attore principale}: Utente
    \item \textbf{Scenario principale}: 
        \begin{enumerate}
            \item L'Utente visualizza l'elenco dei comandi in una pagina separata da quella corrente
        \end{enumerate}
    \item \textbf{Precondizioni}: 
        \begin{itemize}
            \item Pagina dei comandi esistente
        \end{itemize}
    \item \textbf{Postcondizioni}: 
        \begin{itemize}
            \item L'Utente visualizza la pagina dei comandi
        \end{itemize}
    \item \textbf{Inclusioni}:
        \begin{itemize}
		    \item \hyperref[uc_41.1]{UC\_41.1}
		    \item \hyperref[uc_41.2]{UC\_41.2}
		    \item \hyperref[uc_41.3]{UC\_41.3}
		    \item \hyperref[uc_41.4]{UC\_41.4}
		    \item \hyperref[uc_41.5]{UC\_41.5}
        \end{itemize}
    \item \textbf{Trigger}: L'Utente vuole visualizzare l'elenco dei comandi
\end{itemize}
