\label{uc_42}
\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_42.png}
	\caption{UC\_42: Visualizzare elenco dei comandi con spiegazione nella chat}
\end{figure}
\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
	\item \textbf{Scenario principale}: 
	\begin{enumerate}
		\item Il Cliente digita \vr{/comandi} nella chat
		\item Il Cliente visualizza l'elenco dei comandi nella chat con relativa spiegazione
	\end{enumerate}
	\item \textbf{Precondizioni}: 
	\begin{itemize}
		\item Il sistema è online
		\item Il Cliente può scrivere nella chat
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il Cliente visualizza l'elenco dei comandi con spiegazione nella chat
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}

		\item \hyperref[uc_42.6]{UC\_42.6}
	\end{itemize}
	\item \textbf{Trigger}: Il Cliente digita \vr{/comandi} nella chat
\end{itemize}
