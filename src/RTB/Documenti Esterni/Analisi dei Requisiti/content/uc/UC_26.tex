\subsubsection{UC\_26: Invio ordine}
\label{uc_26}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_26}
	\caption{UC\_26: Invio Ordine}
\end{figure}

\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente digita il comando \verb|/invio| nella chat $\rightarrow$ Vedi \hyperref[uc_26.1]{UC\_26.1}
		\item Il Cliente visualizza il riepilogo dell'ordine $\rightarrow$ Vedi \hyperref[uc_26.2]{UC\_26.2}
		\item Il Cliente conferma l'invio dell'ordine $\rightarrow$ Vedi \hyperref[uc_26.3]{UC\_26.3}
		\item Il Cliente visualizza la conferma dell'acquisto $\rightarrow$ Vedi \hyperref[uc_26.4]{UC\_26.4}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il Cliente è nella chat
		\item Sono presenti elementi nel carrello
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item L'ordine viene confermato e inserito nello storico
        \item Il carrello viene svuotato
		\item Il Cliente riceve conferma dell'invio
	\end{itemize}
	\item \textbf{Scenari alternativi}:
	\begin{itemize}
		\item Annullamento ordine a procedura avviata $\rightarrow$ Vedi \hyperref[uc_26.5]{UC\_26.5}
		\item Tentativo di invio di un ordine nullo $\rightarrow$ Vedi \hyperref[uc_27]{UC\_27}
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_26.1]{UC\_26.1}
		\item \hyperref[uc_26.2]{UC\_26.2}
		\item \hyperref[uc_26.3]{UC\_26.3}
		\item \hyperref[uc_26.4]{UC\_26.4}
	\end{itemize}
	\item \textbf{Estensioni}:
	\begin{itemize}
		\item \hyperref[uc_26.5]{UC\_26.5}
		\item \hyperref[uc_27]{UC\_27}
	\end{itemize}
	\item \textbf{Trigger}: Il Cliente deve poter confermare e inviare l'ordine con un semplice comando, per completare l'acquisto in modo rapido.
\end{itemize}

\subsubsubsection{UC\_26.1: Digita comando /invio}
\label{uc_26.1}
\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
	\item \textbf{Scenario principale}: 
	\begin{enumerate}
		\item Il Cliente invia il comando \verb|/invio|
	\end{enumerate}
	\item \textbf{Precondizioni}: 
	\begin{itemize}
		\item Il Cliente si trova nella chat
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il comando \verb|/invio| è stato inviato alla chat
	\end{itemize} 
\end{itemize}

\subsubsubsection{UC\_26.2: Visualizza riepilogo ordine}
\label{uc_26.2}

\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
    \item \textbf{Scenario principale}:
    \begin{enumerate}
        \item Per ogni articolo presente nel carrello, il sistema ne visualizza le informazioni $\rightarrow$ Vedi \hyperref[uc_19.1]{UC\_19.1}
        \item Per ciascun articolo, il Cliente visualizza:
        \begin{itemize}
            \item Il prezzo $\rightarrow$ Vedi \hyperref[uc_26.2.1]{UC\_26.2.1}
            \item Il nome $\rightarrow$ Vedi \hyperref[uc_26.2.2]{UC\_26.2.2}
            \item La quantità $\rightarrow$ Vedi \hyperref[uc_26.2.3]{UC\_26.2.3}
        \end{itemize}
    \end{enumerate}
    \item \textbf{Precondizioni}:
    \begin{itemize}
        \item Il carrello non è vuoto.
        \item Il Cliente ha digitato il comando \verb|/invio| e il sistema sta mostrando il riepilogo.
    \end{itemize}
    \item \textbf{Postcondizioni}:
    \begin{itemize}
        \item Le informazioni di ogni articolo nel carrello sono state visualizzate nel riepilogo.
    \end{itemize}
    \item \textbf{Inclusioni}:
	\begin{itemize}
        \item \hyperref[uc_19.1]{UC\_19.1}
	\end{itemize}
    \item \textbf{Trigger}: Il Cliente vuole visualizzare i dettagli del singolo articolo
\end{itemize}

\subsubsubsection{UC\_26.3: Conferma invio ordine}
\label{uc_26.3}
\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
    \item \textbf{Scenario principale}:
    \begin{enumerate}
        \item Il Cliente conferma l'invio dell'ordine tramite uno dei seguenti metodi:
        \begin{itemize}
            \item Inserendo un comando testuale (es. \vr{conferma}).
            \item Fornendo un comando vocale.
            \item Selezionando un apposito pulsante di conferma nel riepilogo.
        \end{itemize}
    \end{enumerate}
    \item \textbf{Precondizioni}:
    \begin{itemize}
        \item Il carrello non è vuoto.
        \item Il riepilogo dell'ordine è visibile al Cliente.
    \end{itemize}
    \item \textbf{Postcondizioni}:
    \begin{itemize}
        \item La conferma dell'ordine è stata inviata al sistema.
    \end{itemize}
\end{itemize}

\subsubsubsection{UC\_26.4: Visualizza conferma ordine}
\label{uc_26.4}
\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
    \item \textbf{Scenario principale}:
    \begin{enumerate}
        \item Il Cliente visualizza la conferma dell'ordine inviato.
    \end{enumerate}
    \item \textbf{Precondizioni}:
    \begin{itemize}
        \item Il Cliente ha confermato l'invio dell'ordine (UC\_26.3).
    \end{itemize}
    \item \textbf{Postcondizioni}:
    \begin{itemize}
        \item Il sistema ha inviato la conferma dell'ordine, comprensiva di identificativo e riepilogo.
    \end{itemize}
\end{itemize}

\subsubsubsection{UC\_26.5: Annullamento ordine}
\label{uc_26.5}
\begin{itemize}
    \item \textbf{Attore Principale}: Cliente
    \item \textbf{Scenario principale}:
    \begin{enumerate}
        \item Il Cliente seleziona l'opzione per annullare l'ordine (tramite pulsante o comando testuale/vocale).
    \end{enumerate}
    \item \textbf{Precondizioni}:
    \begin{itemize}
        \item Il carrello non è vuoto.
        \item Il riepilogo dell'ordine è visualizzato (procedura di invio avviata).
    \end{itemize}
    \item \textbf{Postcondizioni}:
    \begin{itemize}
        \item La procedura di invio ordine viene interrotta.
        \item Il sistema visualizza un messaggio di conferma dell'annullamento.
        \item Il contenuto del carrello rimane invariato.
    \end{itemize}
\end{itemize}
