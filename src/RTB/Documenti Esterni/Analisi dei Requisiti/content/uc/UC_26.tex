\subsubsection{UC\_26: Invio ordine}
\label{uc_26}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_26}
	\caption{UC\_26: Invio Ordine}
\end{figure}

\begin{itemize}
	\item \textbf{Attore principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente digita il comando \verb|/invio| nella chat $\rightarrow$ Vedi \hyperref[uc_26.1]{UC\_26.1}
		\item Il Cliente visiona il riepilogo dell'ordine $\rightarrow$ Vedi \hyperref[uc_26.2]{UC\_26.2}
		\item Il Cliente conferma l'invio dell'ordine $\rightarrow$ Vedi \hyperref[uc_26.3]{UC\_26.3}
		\item Il Cliente visualizza la conferma dell'acquisto $\rightarrow$ Vedi \hyperref[uc_26.4]{UC\_26.4}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il Cliente è nella chat
		\item Sono presenti elementi nel carrello
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item L'ordine viene confermato e inserito nello storico
		\item Il Cliente riceve conferma dell'invio
	\end{itemize}
	\item \textbf{Scenari alternativi}:
	\begin{itemize}
		\item Annullamento ordine a procedura avviata $\rightarrow$ Vedi \hyperref[uc_26.5]{UC\_26.5}
		\item Tentativo di invio di un ordine nullo $\rightarrow$ Vedi \hyperref[uc_27]{UC\_27}
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_26.1]{UC\_26.1}
		\item \hyperref[uc_26.2]{UC\_26.2}
		\item \hyperref[uc_26.3]{UC\_26.3}
		\item \hyperref[uc_26.4]{UC\_26.4}
	\end{itemize}
	\item \textbf{Estensioni}:
	\begin{itemize}
		\item \hyperref[uc_26.5]{UC\_26.5}
		\item \hyperref[uc_27]{UC\_27}
	\end{itemize}
	\item \textbf{Trigger}: Il Cliente deve poter confermare e inviare l'ordine con un semplice comando, per completare l'acquisto in modo rapido.
\end{itemize}

\subsubsubsection{UC\_26.2: Visualizza riepilogo ordine}
\label{uc_26.2}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_26.2}
	\caption{Inclusioni UC\_26.2: UC\_26.2.1, UC\_26.2.2, UC\_26.2.3}
\end{figure}

\begin{itemize}
    \item \textbf{Attore principale}: Cliente
    \item \textbf{Scenario principale}:
    \begin{enumerate}
        \item Il Cliente vede il riepilogo dell'ordine con un numero identificativo
		\item Il Cliente visualizza il prezzo dei prodotti presenti nel carrello $\rightarrow$ Vedi \hyperref[uc_26.1.1]{UC\_26.1.1}
		\item Il Cliente visualizza il nome dei prodotti presenti nel carrello $\rightarrow$ Vedi \hyperref[uc_26.1.2]{UC\_26.1.2}
		\item Il Cliente visualizza la quantità dei prodotti presenti nel carrello $\rightarrow$ Vedi \hyperref[uc_26.1.3]{UC\_26.1.3}
	\end{enumerate}
    \item \textbf{Precondizioni}:
    \begin{itemize}
        \item Il carrello non è vuoto.
        \item Il Cliente ha digitato \verb|/invio|
    \end{itemize}
    \item \textbf{Postcondizioni}:
    \begin{itemize}
        \item Il riepilogo dell'ordine è visualizzato correttamente.
    \end{itemize}
    \item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_26.2.1]{UC\_26.2.1}
		\item \hyperref[uc_26.2.2]{UC\_26.2.2}
		\item \hyperref[uc_26.2.3]{UC\_26.2.3}
	\end{itemize}
    \item \textbf{Trigger}:
	Il Cliente vuole inviare l'ordine ma è necessario che prima visioni il riepilogo, comprensivo di nomi, quantità e prezzi degli articoli, per verificarne il contenuto.
\end{itemize}

\subsubsubsection{UC\_26.5: Annullamento ordine}
\label{uc_26.5}
\begin{itemize}
    \item \textbf{Attore principale}: Cliente
    \item \textbf{Scenario principale}:
    \begin{enumerate}
        \item Il Cliente annulla l'ordine tramite pulsante.
        \item Il Cliente annulla l'ordine tramite input.
        \item Il Cliente visiona un messaggio di conferma di annullamento dell'invio dell'ordine
    \end{enumerate}
    \item \textbf{Precondizioni}:
    \begin{itemize}
        \item Il carrello non è vuoto.
        \item Il riepilogo dell'ordine è visualizzato.
    \end{itemize}
    \item \textbf{Postcondizioni}:
    \begin{itemize}
        \item L'ordine non viene inviato.
        \item Il contenuto del carrello rimane invariato.
    \end{itemize}
\end{itemize}
