\subsubsubsection{CheckUserController}
\begin{figure}[H]
	\centering 
	\includegraphics[width=1\textwidth]{SpecificaTecnica/Autenticazione/CheckUserController}
	\caption{Autenticazione e Registrazione: CheckUserController}

\paragraph{Descrizione}
\textit{CheckUserController} è il controller REST$^G$ responsabile$^G$ della gestione delle richieste HTTP di autenticazione.\\
Riceve le credenziali dell'utente, le delega al \textit{CheckUserService} e restituisce la risposta appropriata al client.

\paragraph{Attributi}
\textit{CheckUserController} non definisce attributi propri.

\paragraph{Metodi e funzioni}
\begin{itemize}
	\item \textbf{create\_token(username: string): string} - rappresenta la funzione utilizzata per generare il token$^G$ JWT a partire dall'username dell'utente autenticato.
	\item \textbf{login(request: Request, payload: UserSchema, service: \newline CheckUserServiceDep, response: Response): LoginResponse} - \newline metodo pubblico che gestisce la richiesta di login.\\
	Riceve i dati dell'utente tramite \textit{payload}, invoca il service per la verifica$^G$ delle credenziali e restituisce una \textit{LoginResponse} contenente il token$^G$ di accesso o un errore.
\end{itemize}

\subsubsubsection{CheckUserService}
\begin{figure}[H]
	\centering 
	\includegraphics[width=0.6\textwidth]{SpecificaTecnica/Autenticazione/CheckUserService}
	\caption{Autenticazione e Registrazione: CheckUserService}
\end{figure}

\paragraph{Descrizione}
\textit{CheckUserService} è il service applicativo responsabile$^G$ della verifica$^G$ delle credenziali durante il processo$^G$ di login.\\
Interagisce con il repository$^G$ tramite la porta \textit{UserRepoPort} per recuperare i dati dell'utente e verificarne la password.

\paragraph{Attributi}
\begin{itemize}
	\item \textbf{port: UserRepoPort} - attributo privato di tipo \textit{UserRepoPort}.
	Porta utilizzata per accedere ai dati degli utenti nel repository.
\end{itemize}

\paragraph{Metodi e funzioni}
\begin{itemize}
	\item \textbf{\_\_init\_\_(port: UserRepoPort)} - metodo pubblico di inizializzazione, riceve e memorizza la porta di accesso al repository$^G$ degli utenti.
	\item \textbf{check\_user(u: User): string} - metodo pubblico che verifica$^G$ l'esistenza dell'utente e la correttezza della password fornita.\\
	Restituisce un token$^G$ di autenticazione in caso di successo, oppure un messaggio di errore in caso di fallimento.
	\item \textbf{\_verify\_password(plain: str, hashed: str): bool} - metodo privato che confronta una password in chiaro con la corrispondente versione hashata.\\
	Restituisce \textit{true} se le due stringhe corrispondono.
\end{itemize}

\subsubsubsection{RegisterUserService}
\begin{figure}[H]
	\centering 
	\includegraphics[width=0.8\textwidth]{SpecificaTecnica/Autenticazione/RegisterUserService}
	\caption{Autenticazione e Registrazione: RegisterUserService}
\end{figure}

\paragraph{Descrizione}
\textit{RegisterUserService} è il service applicativo responsabile$^G$ della registrazione di nuovi utenti.\\
Verifica$^G$ la disponibilità di username ed email, valida il dominio dell'indirizzo email e persiste il nuovo utente tramite \textit{UserRepoPort} previa cifratura della password.

\paragraph{Attributi}
\begin{itemize}
	\item \textbf{port: UserRepoPort} - attributo privato di tipo \textit{UserRepoPort}.\\
	Porta utilizzata per accedere al repository$^G$ degli utenti e verificare la disponibilità delle credenziali e salvare il nuovo utente.
\end{itemize}

\paragraph{Metodi e funzioni}
\begin{itemize}
	\item \textbf{\_\_init\_\_(port: IUserRepoPort)} - metodo pubblico di inizializzazione.\\
	Riceve e memorizza la porta del repository$^G$.
	\item \textbf{register\_user(u: UserRegistration)} - metodo pubblico che gestisce il flusso completo di registrazione: verifica$^G$ la disponibilità di username ed email, valida il dominio email, cifra la password e salva il nuovo utente tramite il repository.
	\item \textbf{\_hash\_password(password: string): string} - metodo privato che cifra la password in chiaro e restituisce la corrispondente stringa hashata da memorizzare nel database.
	\item \textbf{domain\_exists(email: string): bool} - metodo pubblico che verifica$^G$ l'esistenza del dominio dell'indirizzo email fornito.\\
\end{itemize}

\subsubsubsection{UserRepoPort}
\begin{figure}[H]
	\centering 
	\includegraphics[width=0.6\textwidth]{SpecificaTecnica/Autenticazione/UserRepoPort}
	\caption{Autenticazione e Registrazione: UserRepoPort}
\end{figure}

\paragraph{Descrizione}
\textit{UserRepoPort} è un'interfaccia che definisce il contratto per l'accesso ai dati degli utenti.\\
Viene implementata da \textit{UserRepoAdapter} e utilizzata dai service del dominio, garantendo il disaccoppiamento tra logica di business e layer di persistenza.

\paragraph{Attributi}
Essendo un'interfaccia, non definisce attributi.

\paragraph{Metodi e funzioni}
\begin{itemize}
	\item \textbf{find\_by\_username(username: string): WebUser} - metodo pubblico che recupera un utente dal repository$^G$ a partire dal suo username.\\
	Restituisce l'oggetto \textit{WebUser} corrispondente.
	\item \textbf{username\_exists(username: string): bool} - metodo pubblico che verifica$^G$ se un determinato username è già presente nel sistema.\\
	Restituisce \textit{true} se l'username esiste, \textit{false} altrimenti.
	\item \textbf{email\_exists(email: string): bool} - metodo pubblico che verifica$^G$ se un determinato indirizzo email è già registrato nel sistema.\\
	Restituisce \textit{true} se l'email esiste, \textit{false} altrimenti.
	\item \textbf{add\_user(u: UserRegistration): bool} - metodo pubblico che aggiunge un nuovo utente al repository$^G$ a partire dai dati di registrazione forniti.\\
	Restituisce \textit{true} in caso di successo.
	\item \textbf{delete\_user(username: string)} - metodo pubblico che rimuove dal repository$^G$ l'utente identificato dall'username fornito.
	\item \textbf{reset\_password(u: UserReset)} - metodo pubblico che aggiorna la password dell'utente utilizzando i dati contenuti nell'oggetto \textit{UserReset}.
\end{itemize}


\subsubsubsection{UserRepoAdapter}
\begin{figure}[H]
	\centering 
	\includegraphics[width=0.4\textwidth]{SpecificaTecnica/Autenticazione/UserRepoAdapter}
	\caption{Autenticazione e Registrazione: UserRepoAdapter}
\end{figure}

\paragraph{Descrizione}
\textit{UserRepoAdapter} è la classe concreta che implementa l'interfaccia \textit{UserRepoPort}, fungendo da adattatore tra il dominio applicativo e il layer di persistenza.\\
Delega le operazioni effettive sul database$^G$ all'istanza di \textit{UserRepository} che incapsula al proprio interno.

\paragraph{Attributi}
\begin{itemize}
	\item \textbf{repo: UserRepository} - attributo privato di tipo \textit{UserRepository}.\\
	Rappresenta il repository$^G$ concreto utilizzato per eseguire le query$^G$ sul database.
\end{itemize}

\paragraph{Metodi e funzioni}
Oltre ad essere implementate tutte le funzioni di \textit{UserRepoPort}, è disponibile anche la seguente funzione:
\begin{itemize}
	\item \textbf{\_\_init\_\_(db$^G$: Session)} - metodo pubblico di inizializzazione.\\
	Riceve una sessione di database$^G$ di tipo \textit{Session} e la utilizza per istanziare il \textit{UserRepository} interno.
\end{itemize}


\subsubsubsection{UserRepository}
\begin{figure}[H]
	\centering 
	\includegraphics[width=0.6\textwidth]{SpecificaTecnica/Autenticazione/UserRepository}
	\caption{Autenticazione e Registrazione: UserRepository}
\end{figure}

\paragraph{Descrizione}
\textit{UserRepository} è la classe che gestisce direttamente le operazioni di accesso e manipolazione dei dati degli utenti sul database.\\
Utilizza un \textit{QueryExecutor} interno per eseguire le query$^G$ in modo centralizzato e uniforme.

\paragraph{Attributi}
\begin{itemize}
	\item \textbf{executor: QueryExecutor} - attributo privato di tipo \textit{QueryExecutor}.
	Componente utilizzato per l'esecuzione delle query$^G$ sul database$^G$, centralizzando la gestione delle operazioni di persistenza.
\end{itemize}

\paragraph{Metodi e funzioni}
\begin{itemize}
	\item \textbf{\_\_init\_\_(db$^G$: Session)} - metodo pubblico di inizializzazione.
	Riceve una sessione di database$^G$ e la utilizza per configurare il \textit{QueryExecutor}.
	\item \textbf{find\_by\_username(username: string): WebUser} - metodo pubblico che esegue una query$^G$ per recuperare l'utente corrispondente all'username fornito.\\
	Restituisce l'oggetto \textit{WebUser} trovato.
	\item \textbf{find\_by\_email(email: string): WebUser} - metodo pubblico che esegue una query$^G$ per recuperare l'utente corrispondente all'indirizzo email fornito.\\
	Restituisce l'oggetto \textit{WebUser} trovato.
	\item \textbf{save(u: UserRegistration): bool} - metodo pubblico che persiste un nuovo utente nel database$^G$ a partire dall'oggetto \textit{UserRegistration} fornito.\\
	Restituisce \textit{true} in caso di successo.
	\item \textbf{delete(username: string): bool} - metodo pubblico che rimuove dal database$^G$ l'utente identificato dall'username fornito.\\
	Restituisce \textit{true} se l'operazione va a buon fine.
	\item \textbf{reset\_password(u: UserReset): bool} - metodo pubblico che aggiorna la password dell'utente nel database$^G$ utilizzando i dati dell'oggetto \textit{UserReset}.\\
	Restituisce \textit{true} in caso di successo.
\end{itemize}
