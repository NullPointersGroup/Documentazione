\subsubsection{UC\_23: Fornire feedback all'AI}
\label{uc_23}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.8\textwidth]{AnalisiRequisiti/UC_23}
	\caption{UC\_23: Fornire feedback all'AI}
\end{figure}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente, tramite il pulsante affianco all'ultimo messaggio dell'AI, apre il form per la compilazione del feedback$^G$
		\item Il Cliente compila il campo tipologia del feedback$^G$ $\rightarrow$ Vedi \hyperref[uc_23.1]{UC\_23.1}
		\item Il Cliente compila il campo descrizione del feedback$^G$ $\rightarrow$ Vedi \hyperref[uc_23.2]{UC\_23.2}
        \item Il Cliente invia il feedback$^G$ $\rightarrow$ Vedi \hyperref[uc_23.3]{UC\_23.3}
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il sistema è online
		\item Il Cliente ha ricevuto una risposta dal chatbot AI
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il feedback$^G$ è stato registrato nel sistema
		\item Il Cliente ha ricevuto conferma dell'operazione
	\end{itemize}
	\item \textbf{Inclusioni}:
	\begin{itemize}
		\item \hyperref[uc_23.1]{UC\_23.1}
		\item \hyperref[uc_23.2]{UC\_23.2}
		\item \hyperref[uc_23.3]{UC\_23.3}
	\end{itemize}
	\item \textbf{Trigger}: Il Cliente vuole valutare le risposte del chatbot tramite feedback$^G$, per contribuire a migliorarne l'accuratezza
\end{itemize}

\subsubsubsection{UC\_23.1: Inserimento tipologia feedback}
\label{uc_23.1}
\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
		\item Il Cliente compila il campo relativo alla tipologia del feedback$^G$
	\end{enumerate}
	\item \textbf{Precondizioni}:
	\begin{itemize}
		\item Il modulo di feedback$^G$ è aperto e il campo tipologia è visibile al Cliente
	\end{itemize}
	\item \textbf{Postcondizioni}:
	\begin{itemize}
		\item Il campo tipologia del feedback$^G$ è compilato
	\end{itemize}
\end{itemize}

\subsubsubsection{UC\_23.3: Invio feedback}

\begin{itemize}
    \item \textbf{Attore$^G$ Principale}: Cliente
	\item \textbf{Scenario principale}:
	\begin{enumerate}
